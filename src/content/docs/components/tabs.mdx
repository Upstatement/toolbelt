## Anatomy

```html
<div x-tabs>
  <div x-tabs:list>
    <button x-tabs:tab></button>
  </div>

  <div x-tabs:panel></div>
</div>
```

## Directives

<DirectiveReference
  directive="x-tabs"
  description="The root element for tabs."
  required
>
  <Fragment slot="value">
    The default tab to be displayed. Otherwise, display the first tab.
  </Fragment>
</DirectiveReference>

<DirectiveReference
  directive="x-tabs:list"
  description="List of tabs."
  required
>
  <Fragment slot="modifiers">

| Modifier                     | Description                                                                      |
| ---------------------------- | -------------------------------------------------------------------------------- |
| `.loop`                      | Loops keyboard navigation (e.g. arrow keys)                                      |
| `.automatic`                 | Automatically show/hide panel with keyboard navigation (e.g. arrow keys)         |
| `.vertical` \| `.horizontal` | Orientation of the accordion for keyboard navigation. Defaults to `.horizontal`. |

  </Fragment>
</DirectiveReference>

<DirectiveReference
  directive="x-tabs:tab"
  description="Trigger button that shows the corresponding panel."
  required
>
  <Fragment slot="value">The value for this tab.</Fragment>

  <Fragment slot="attributes">

| Data Attribute | Values                 |
| -------------- | ---------------------- |
| `[data-state]` | `active` \| `inactive` |

  </Fragment>
</DirectiveReference>

<DirectiveReference
  directive="x-tabs:panel"
  description="Panel that contains content."
  required
>
  <Fragment slot="value">
    The value for this panel, which should match the correct tab.
  </Fragment>

  <Fragment slot="attributes">

| Data Attribute | Values                 |
| -------------- | ---------------------- |
| `[data-state]` | `active` \| `inactive` |

  </Fragment>
</DirectiveReference>
