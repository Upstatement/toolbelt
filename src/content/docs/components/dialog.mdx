## Anatomy

```html
<div x-dialog>
  <button x-dialog:trigger></button>

  <template x-teleport="body">
    <div x-dialog:content>
      <div x-dialog:overlay></div>
      <h2 x-dialog:title></h2>
      <p x-dialog:description></p>
      <button x-dialog:close></button>
    </div>
  </template>
</div>
```

## Directives

<DirectiveReference
  directive="x-dialog"
  description="The root element for the dialog."
  required
/>

<DirectiveReference
  directive="x-dialog:trigger"
  description="Trigger button that opens or closes the dialog content."
  required
>
  <Fragment slot="attributes">

| Data Attribute | Values             |
| -------------- | ------------------ |
| `[data-state]` | `open` \| `closed` |

  </Fragment>
</DirectiveReference>

<DirectiveReference
  directive="x-dialog:content"
  description="The content that will be shown or hidden."
  required
>
  <Fragment slot="attributes">

| Data Attribute | Values             |
| -------------- | ------------------ |
| `[data-state]` | `open` \| `closed` |

  </Fragment>
</DirectiveReference>

<DirectiveReference
  directive="x-dialog:overlay"
  description="Overlay for the page content. Closes the dialog when clicked on."
/>

<DirectiveReference
  directive="x-dialog:title"
  description="Dialog title announced to screen readers when opened."
/>

<DirectiveReference
  directive="x-dialog:description"
  description="Dialog description announced to screen readers when opened."
/>

<DirectiveReference
  directive="x-dialog:close"
  description="Button to close the dialog when opened."
  required
/>
