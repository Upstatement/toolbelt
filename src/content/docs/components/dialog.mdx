## Anatomy

```html
<div x-dialog>
  <button x-dialog:trigger></button>

  <template x-teleport="body">
    <div x-dialog:content>
      <div x-dialog:overlay></div>
      <h2 x-dialog:title></h2>
      <p x-dialog:description></p>
      <button x-dialog:close></button>
    </div>
  </template>
</div>
```

## Directives

<DirectiveReference directive="x-dialog" required>
  The root element for the dialog.
</DirectiveReference>

<DirectiveReference
  directive="x-dialog:trigger"
  required
>
  Trigger button that opens or closes the dialog content.

  <Table>

| Data Attribute | Values             |
| -------------- | ------------------ |
| `[data-state]` | `open` \| `closed` |

  </Table>
</DirectiveReference>

<DirectiveReference
  directive="x-dialog:content"
  required
>
  The content that will be shown or hidden.

  <Table>

| Data Attribute | Values             |
| -------------- | ------------------ |
| `[data-state]` | `open` \| `closed` |

  </Table>
</DirectiveReference>

<DirectiveReference directive="x-dialog:overlay">
  Overlay for the page content. Closes the dialog when clicked on.
</DirectiveReference>

<DirectiveReference directive="x-dialog:title">
  Dialog title announced to screen readers when opened.
</DirectiveReference>

<DirectiveReference directive="x-dialog:description">
  Dialog description announced to screen readers when opened.
</DirectiveReference>

<DirectiveReference directive="x-dialog:close" required>
  Button to close the dialog when opened.
</DirectiveReference>
